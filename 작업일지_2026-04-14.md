### 백필 스크립트 작성 방법

아래 스크립트를 `scripts/backfill_flow.py`로 만들어 실행:

```python
"""
수급 데이터 백필 스크립트.
누락된 날짜의 수급(외인/기관/개인 순매수) 데이터를 KIS API로 재수집.
실행: python scripts/backfill_flow.py
"""
import sys, os, time
from pathlib import Path
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from dotenv import load_dotenv
load_dotenv()

import requests
from sqlalchemy import text
from sqlalchemy.dialects.postgresql import insert as pg_insert

from backend.collector.spot import _get_kis_token
from backend.db.database import SessionLocal
from backend.db.models import SpotInvestorFlow

token = _get_kis_token()
app_key = os.getenv("KIS_APP_KEY", "")
app_secret = os.getenv("KIS_APP_SECRET", "")

h = {
    'authorization': f'Bearer {token}',
    'appkey': app_key,
    'appsecret': app_secret,
    'tr_id': 'FHKST01010900',
    'content-type': 'application/json',
}

db = SessionLocal()

# 누락된 날짜+종목 조합 조회
missing = db.execute(text("""
    SELECT f.trading_date, f.stock_code
    FROM spot_investor_flows f
    JOIN spot_daily_prices p
      ON p.trading_date = f.trading_date AND p.stock_code = f.stock_code
    WHERE f.foreign_net_buy = 0
      AND f.institution_net_buy = 0
      AND f.individual_net_buy = 0
    ORDER BY f.trading_date, f.stock_code
""")).fetchall()

print(f"총 누락: {len(missing)}건")

updated = 0
for i, (tdate, code) in enumerate(missing):
    yyyymmdd = tdate.strftime('%Y%m%d')
    try:
        r = requests.get(
            'https://openapi.koreainvestment.com:9443/uapi/domestic-stock/v1/quotations/inquire-investor',
            headers=h,
            params={'fid_cond_mrkt_div_code': 'J', 'fid_input_iscd': code},
            timeout=8,
        )
        rows = r.json().get('output', [])
        hit = next((row for row in rows if row.get('stck_bsop_date') == yyyymmdd), None)
        if hit:
            f_val = float(hit.get('frgn_ntby_qty') or 0)
            i_val = float(hit.get('orgn_ntby_qty') or 0)
            p_val = float(hit.get('prsn_ntby_qty') or 0)
            if f_val != 0 or i_val != 0 or p_val != 0:
                db.execute(
                    pg_insert(SpotInvestorFlow).values(
                        trading_date=tdate, stock_code=code,
                        foreign_net_buy=f_val, institution_net_buy=i_val, individual_net_buy=p_val
                    ).on_conflict_do_update(
                        constraint='uq_spot_investor_flows',
                        set_={'foreign_net_buy': f_val, 'institution_net_buy': i_val, 'individual_net_buy': p_val}
                    )
                )
                updated += 1
    except Exception as e:
        pass

    if (i + 1) % 100 == 0:
        db.commit()
        print(f"  [{i+1}/{len(missing)}] updated={updated}")
    
    time.sleep(0.12)

db.commit()
db.close()
print(f"\n완료: {updated}/{len(missing)}건 업데이트")
```
